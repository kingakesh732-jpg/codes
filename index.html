#!/bin/bash # ========================================================= # Script: NeoPlayz All-In-One (RE-FIXED) # Credits: Original Logic by Jishnu # Modified & Debugged for: codes.neoplayz.site # ========================================================= # Setup Colors RED='\e[1;31m' GREEN='\e[1;32m' YELLOW='\e[1;33m' BLUE='\e[1;34m' CYAN='\e[1;36m' NC='\e[0m' clear # NeoPlayz Branding Banner echo -e "${CYAN}==========================================${NC}" echo -e "${GREEN} NEOPLAYZ VPS SYSTEM (STABLE) ${NC}" echo -e "${CYAN}==========================================${NC}" # 1. Dependency Check check_requirements() { echo -e "${BLUE}[*] Checking Requirements...${NC}" sudo apt update -y && sudo apt install git curl nodejs npm -y --quiet } # 2. User Creation Logic create_neouser() { echo -e "${BLUE}[*] Setting up NeoPlayz User...${NC}" if id "neouser" &>/dev/null; then echo -e "${YELLOW}[!] User 'neouser' already exists.${NC}" else sudo useradd -m -s /bin/bash neouser echo "neouser:password123" | sudo chpasswd echo -e "${GREEN}[+] User 'neouser' created successfully.${NC}" fi } # 3. Panel Installation Logic install_neopanel() { echo -e "${BLUE}[*] Downloading Panel...${NC}" mkdir -p ~/neoplayz_vps && cd ~/neoplayz_vps sudo rm -rf ./* .git git clone https://github.com/bravo68web/cloud-player.git . --quiet npm install --quiet sudo npm install -g pm2 --quiet pm2 delete "neoplayz-panel" &>/dev/null pm2 start index.js --name "neoplayz-panel" echo -e "${GREEN}[+] Panel is now running on PM2!${NC}" } # 4. The Master Function (Option 1) full_setup() { check_requirements create_neouser install_neopanel echo -e "\n${GREEN}✅ EVERYTHING INSTALLED SUCCESSFULLY!${NC}" } # --- Main Menu Loop --- while true; do echo -e "\n${YELLOW}NeoPlayz Menu - Choose an Option:${NC}" echo -e "1) 🚀 Full Setup (Jishnu Style)" echo -e "2) 👤 Create User Only" echo -e "3) 🌐 Install Panel Only" echo -e "4) ⚙️ Restart All Services" echo -e "5) ❌ Exit" echo -ne "${CYAN}Selection: ${NC}" read opt case $opt in 1) full_setup ;; 2) create_neouser ;; 3) install_neopanel ;; 4) pm2 restart all || echo -e "${RED}PM2 not running.${NC}" ;; 5) echo -e "${GREEN}Goodbye!${NC}"; exit 0 ;; *) echo -e "${RED}Invalid Option! Try again.${NC}" ;; esac echo -e "${CYAN}------------------------------------------${NC}" done
